    document.getElementById("nodo-principal").addEventListener("click", function() {
        const container = document.querySelector(".mapa-container");
        let nodos = document.querySelectorAll(".nodo-hijo");

        if (nodos.length === 0) {
            const elementos = [
                {texto: "Salud", icono: "https://cdn-icons-png.freepik.com/512/3182/3182127.png"},
                {texto: "Seguridad", icono: "https://cdn-icons-png.freepik.com/512/1022/1022382.png"},
                {texto: "Medio Ambiente", icono: "https://cdn-icons-png.freepik.com/512/1598/1598238.png"},
                {texto: "Educación", icono: "https://cdn-icons-png.freepik.com/512/3976/3976555.png"},
                {texto: "Innovación", icono: "https://cdn-icons-png.freepik.com/512/1087/1087927.png"},
                {texto: "Tecnología", icono: "https://cdn-icons-png.freepik.com/512/780/780477.png"},
                {texto: "Cultura", icono: "https://cdn-icons-png.freepik.com/512/2788/2788019.png"},
                {texto: "Sostenibilidad", icono: "https://cdn-icons-png.freepik.com/512/12134/12134560.png"}
            ];

            elementos.forEach((elem, i) => {
                let nodo = document.createElement("div");
                nodo.classList.add("nodo", "nodo-hijo");
                nodo.innerHTML = `<img src="${elem.icono}" alt="${elem.texto}"><span>${elem.texto}</span>`;
                container.appendChild(nodo);

                let angle = (i / elementos.length) * (2 * Math.PI);
                let x = 200 * Math.cos(angle);
                let y = 200 * Math.sin(angle);
                nodo.style.transform = `translate(${x}px, ${y}px)`;
                nodo.style.opacity = "1";
                nodo.style.display = "flex";
            });
        } else {
            nodos.forEach(nodo => nodo.remove());
        }
    });
</script>
